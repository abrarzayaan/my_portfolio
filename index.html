-- Optimized Query for Sales Analysis
WITH sales_cte AS (
    SELECT 
        customer_id,
        SUM(order_amount) AS total_spent,
        COUNT(*) AS order_count,
        MAX(order_date) AS last_order
    FROM sales_fact s
    JOIN date_dim d ON s.date_key = d.date_key
    WHERE d.year = 2023
        AND d.quarter IN (3, 4)
    GROUP BY customer_id
    HAVING SUM(order_amount) > 1000
)
SELECT 
    c.customer_name,
    c.segment,
    sc.total_spent,
    sc.order_count,
    DATEDIFF(day, sc.last_order, GETDATE()) AS days_since_last_order
FROM customer_dim c
JOIN sales_cte sc ON c.customer_id = sc.customer_id
ORDER BY sc.total_spent DESC;
